437 def attachment(filename = nil, disposition='attachment')
438   if filename
439     param_filename = File.basename(filename)
440     encoding = param_filename.encoding
441 
442     needs_encoding = param_filename.gsub!(/[^ 0-9a-zA-Z!\#$&\+\.\^_`\|~]+/, '-')
443     params = "; filename=#{param_filename.inspect}"
444 
445     if needs_encoding && (encoding == UTF8_ENCODING || encoding == ISO88591_ENCODING)
446       # File name contains non attr-char characters from RFC 5987 Section 3.2.1
447 
448       encoded_filename = File.basename(filename).force_encoding(BINARY_ENCODING)
449       # Similar regexp as above, but treat each byte separately, and encode
450       # space characters, since those aren't allowed in attr-char
451       encoded_filename.gsub!(/[^0-9a-zA-Z!\#$&\+\.\^_`\|~]/) do |c|
452         "%%%X" % c.ord
453       end
454 
455       encoded_params = "; filename*=#{encoding.to_s}''#{encoded_filename}"
456     end
457 
458     unless @headers["Content-Type"]
459       ext = File.extname(filename)
460       unless ext.empty?
461         content_type(ext)
462       end
463     end
464   end
465   @headers["Content-Disposition"] = "#{disposition}#{params}#{encoded_params}"
466 end

419 def content_type(type = nil || (return @headers["Content-Type"]), opts = OPTS)
420   unless (mime_type = mime_type(type) || opts[:default])
421     raise RodaError, "Unknown media type: #{type}"
422   end

424   unless opts.empty?
