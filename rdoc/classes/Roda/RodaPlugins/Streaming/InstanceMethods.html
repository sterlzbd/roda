def stream(opts=OPTS, &block)
  if opts[:loop]
    block = proc do |out|
      until out.closed?
        begin
          yield(out)
        rescue => e
          handle_stream_error(e, out)
        end
      end
    end
  end

  throw :halt, @_response.finish_with_body(Stream.new(opts, &block))
end
